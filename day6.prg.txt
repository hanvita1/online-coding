void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}

void permute(char *a, int i, int n)
{
    int j;
    if (i == n)
        printf("%s\n", a);
    else {
        for (j = i; j <= n; j++)
        {
            swap((a + i), (a + j));
            permute(a, i + 1, n);
            swap((a + i), (a + j)); 
        }
    }
}

}

void cntArray(int A[], int N) 
{ 
 
    int result = 0; 
  
    for (int i = 0; i < N; i++) { 
          result++; 
        int current_value = A[i]; 
  
        for (int j = i + 1; j < N; j++) { 
  
            if (A[j] == current_value) { 
                result++; 
            } 
        } 
    } 

    cout << result << endl; 
}
